; Stores result back to X and halts

        LD R0, X          ; Load X from memory
        SHL R0            ; Logical shift left by 1 bit
        SHL R0            ; Shift left again (total of 2 bits)

        ; Clear top 2 bits using mask 0xFC (11111100)
        LDI R1, 0xFC      ; Load bitmask
        AND R0, R1        ; Keep only lower 6 bits

        ST R0, X          ; Store result back to X

loop:   JMP loop          ; Infinite loop to end program
